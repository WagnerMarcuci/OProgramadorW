#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX 10

int comparar (const void * a, const void * b)
{
    return ( *(int*)a - *(int*)b );
}

int main()
{
    char cValor[ 10 + 1 ] = "";
    int valor[ MAX ];
    int procurar = 0;
    int metade = 0;
    int inicio = 0;
    int fim = 0;

    memset( &valor, '\0', sizeof( int ) * MAX );

    for( int i = 0; i < MAX; ++i )
    {
        printf( "entre com o %do. numero: ", i + 1 );
        fgets( cValor, sizeof( cValor ), stdin );

        valor[ i ] = atoi( cValor );
    }

    printf( "\n\nIMPRIMIR ENTRADA\n" );

    for( int i = 0; i < MAX; ++i )
    {
        printf( "%d - %d\n", i + 1, valor[ i ] );
    }

    qsort (valor, MAX, sizeof (int), comparar);

    printf( "\n\nIMPRIMIR ORGANIZADO\n" );

    for( int i = 0; i < MAX; ++i )
    {
        printf( "%d - %d\n", i, valor[ i ] );
    }

    printf( "\n\nprocurar pelo numero: " );
    fgets( cValor, sizeof( cValor ), stdin );
    procurar = atoi( cValor );
    inicio = 0;
    fim = sizeof( valor ) / sizeof( valor[ 0 ] );
    fim -= 1;

    while( 1 )
    {
        metade = ( inicio + fim ) / 2;
        printf( "inicio: %d, fim: %d = metade: %d\n", inicio, fim, metade );

        if ( procurar == valor[ metade ] )
        {
            printf( "valor no indice %d\n", metade );
            break;
        }
        else if ( procurar < valor[ metade ] )
        {
            fim = metade - 1;
            continue;
        }
        else if( procurar > valor[ metade ] )
        {
            inicio = metade + 1;
        }
        else
        {
            printf( "busca finalizada, nao encontrado.\n" );
            break;
        }

    }

    return 0;
}
