#include <gtk/gtk.h>
#include <stdio.h>

int main( int argc, char * argv[] )
{
    gtk_init( &argc, &argv );
    
    GtkBuilder * builder = gtk_builder_new_from_file( "teste_botao.ui" );
    GtkWidget * layout = GTK_WIDGET( gtk_builder_get_object( builder, "layout_principal" ) );
    GtkWidget * janela = GTK_WIDGET( gtk_builder_get_object( builder, "janela_principal" ) );

    g_signal_connect( janela, "destroy", G_CALLBACK( gtk_main_quit ), NULL );

    gtk_builder_connect_signals( builder, NULL );

    GtkWidget * botao_01 = GTK_WIDGET( gtk_builder_get_object( builder, "botao_01" ) );

    gtk_widget_show( janela );

    gtk_main();
    
    return 0;
}
