#include <gtk/gtk.h>
#include <stdio.h>

GtkWidget * g_radio = NULL;
GtkWidget * g_radio1 = NULL;
GtkWidget * g_radio2 = NULL;

GtkWidget * g_radio11 = NULL;
GtkWidget * g_radio12 = NULL;

void eventos_checkbox( GtkWidget * widget, gpointer * data)
{
   printf( "evento: %s\n", data);
   
   if ( gtk_toggle_button_get_active( GTK_TOGGLE_BUTTON( g_radio ) ) )
   {
       printf( "radio checado.\n" );
   }
   else
   {
       printf( "radio NAO checado.\n" );
   }
}

void criar_radiobox( void )
{
    g_radio = gtk_radio_button_new_with_label( NULL, "radio" );

    // criacao de um grupo de radiobox
    GSList * grupo_radiobox = gtk_radio_button_get_group( GTK_RADIO_BUTTON( g_radio ) );
    
    // radiobox1 agora faz parte do grupo
    g_radio1 = gtk_radio_button_new_with_label( grupo_radiobox, "radio 1" );
    
    // temos que obter o grupo novamente com todos os botoes
    grupo_radiobox = gtk_radio_button_get_group( GTK_RADIO_BUTTON( g_radio ) );

    // radiobox2 agora faz parte do grupo
    g_radio2 = gtk_radio_button_new_with_label( grupo_radiobox, "radio 2" );
}

void criar_radiobox_outro_grupo( void )
{
    g_radio11 = gtk_radio_button_new_with_label( NULL, "radio-11" );

    // criacao de um grupo de radiobox
    GSList * grupo_radiobox = gtk_radio_button_get_group( GTK_RADIO_BUTTON( g_radio11 ) );
    
    // radiobox12 agora faz parte do grupo
    g_radio12 = gtk_radio_button_new_with_label( grupo_radiobox, "radio 12" );
}

int main( int argc, char * argv[] )
{
    GtkWidget * janela = NULL;
    
    gtk_init( &argc, &argv );
    
    janela = gtk_window_new( GTK_WINDOW_TOPLEVEL );
    
    g_signal_connect( janela, "destroy", G_CALLBACK( gtk_main_quit ), NULL );
    
    gtk_widget_show( janela );
    
    gtk_window_set_title( ( GtkWindow * )janela, "Titulo da Janela" );
    
    gtk_window_resize( ( GtkWindow * )janela, 400, 500 );
    
    criar_radiobox();

    criar_radiobox_outro_grupo();
    
    // ja esta clicado
    //gtk_toggle_button_set_active(GTK_TOGGLE_BUTTON( g_radio ), TRUE);

    // eventos
   g_signal_connect ( g_radio, "toggled",
			G_CALLBACK ( eventos_checkbox ), "check 1 toggled");
   g_signal_connect ( g_radio, "pressed",
			G_CALLBACK ( eventos_checkbox ), "check 1 pressed");
   g_signal_connect ( g_radio, "released",
			G_CALLBACK ( eventos_checkbox ), "check 1 released");
    g_signal_connect ( g_radio, "clicked",
			G_CALLBACK ( eventos_checkbox ), "check 1 clicked");
   g_signal_connect ( g_radio, "enter",
			G_CALLBACK ( eventos_checkbox ), "check 1 enter");
   g_signal_connect ( g_radio, "leave",
			G_CALLBACK ( eventos_checkbox ), "check 1 leave");    
    
    GtkWidget * h1 = gtk_box_new( GTK_ORIENTATION_VERTICAL, 5 );
    
    gtk_box_pack_start( GTK_BOX( h1 ), g_radio, FALSE, FALSE, 0 );
    gtk_box_pack_start( GTK_BOX( h1 ), g_radio1, FALSE, FALSE, 0 );
    gtk_box_pack_start( GTK_BOX( h1 ), g_radio2, FALSE, FALSE, 0 );
    
    gtk_box_pack_start( GTK_BOX( h1 ), g_radio11, FALSE, FALSE, 0 );
    gtk_box_pack_start( GTK_BOX( h1 ), g_radio12, FALSE, FALSE, 0 );
    
    gtk_container_add( GTK_CONTAINER( janela ), h1 );
    
    gtk_widget_show_all( janela );
    
    gtk_main();
    
    return 0;
}
