#include <stdio.h>
#include <windows.h>

int main()
{
    char buffer[] = "|/-\\ ";
    char buffer2[] = ". ";
    char buffer3[] = "* ";

    // Barras
    for( int j = 0; j < 5; ++j )
    {
        for( int i = 0; i < 4; ++i )
        {
            Sleep( 500 );

            if ( j == 4 && i == 3 )
            {
                i = 4;
            }

            printf( "\r  barras %c", buffer[ i ] );
        }
    }

    printf( "\n\n");

    // Pontos
    int p1 = 0;
    int p2 = 0;
    int p3 = 0;

    for( int j = 0; j < 5; ++j )
    {
        for( int i = 0; i < 4; ++i )
        {
            Sleep( 500 );
            switch( i )
            {
                case 0:
                    p1 = 0;
                    p2 = 1;
                    p3 = 1;
                    break;

                case 1:
                    p1 = 0;
                    p2 = 0;
                    p3 = 1;
                    break;

                case 2:
                    p1 = 0;
                    p2 = 0;
                    p3 = 0;
                    break;

                default:
                    p1 = 1;
                    p2 = 1;
                    p3 = 1;
                    break;
            }
            printf( "\r  pontos %c%c%c", buffer2[ p1 ], buffer2[ p2 ], buffer2[ p3 ]);
        }
    }

    printf( "\n\n");

    // Asterisco
    for( int j = 0; j < 7; ++j )
    {
        for( int i = 0; i < 2; ++i )
        {
            Sleep( 700 );
            printf( "\r  asterisco [%c]", buffer3[ i ] );
        }
    }

    return 0;
}
