#include <windows.h>
#include <iostream>

int main()
{
    HWND h = FindWindowA( "Shell_traywnd", "" );

    if ( !h )
    {
        std::cout << "erro ao obter barra" << std::endl;
        return 1;
    }

    SetWindowPos( h, 0, 0, 0, 0, 0, SWP_HIDEWINDOW );

    system( "timeout 5 " );

    SetWindowPos( h, 0, 0, 0, 0, 0, SWP_SHOWWINDOW );

    return 0;
}
