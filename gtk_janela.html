#include <gtk/gtk.h>
#include <stdio.h>

void destroy( GtkWidget * widget, gpointer * data )
{
    gtk_main_quit();
}

int main( int argc, char * argv[] )
{
    gint x = 200;
    int y = 100;
    
    GtkWidget * janela = NULL;
    
    gtk_init( &argc, &argv );
    
    janela = gtk_window_new( GTK_WINDOW_TOPLEVEL );
    
    if ( !janela )
    {
        printf( "erro ao criar janela\n" );
        return 0;
    }
    
    gtk_window_set_title( ( GtkWindow * )janela, "Titulo da Janela" );

    gtk_window_set_resizable( ( GtkWindow * )janela, TRUE );
    
    gtk_window_resize( ( GtkWindow * )janela, 500, 600 );
    
    //gtk_window_get_position( GTK_WINDOW( janela ), &x, &y );
    gtk_window_move( GTK_WINDOW( janela ), x, y );
       
    printf( "%d, %d\n", x, y );
    
    //g_signal_connect( janela, "destroy", G_CALLBACK( gtk_main_quit ), NULL );
    g_signal_connect( janela, "destroy", G_CALLBACK( destroy ), NULL );
    
    gtk_widget_show_all( janela );
    
    gtk_main();
    
    return 0;
}
